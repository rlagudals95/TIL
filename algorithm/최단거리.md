function solution(maps) {
    let answer = -1;
    
    const len = maps[0].length - 1;
    
    const dx = [-1, 0, 1, 0];
    const dy = [0, 1, 0, -1];

    
    function DFS (count, x, y) {
        
        if(count === -1 && count > answer){
            return
        }
        
        if(x === len && y === len){
            if(answer === - 1){
                answer = count
            } else {
                answer = Math.min(answer, count)
            }
        } else {
            for (let i = 0; i < 4; i++){
                const nx = x + dx[i];
                const ny = y + dy[i];
                
                if(nx >= 0 && ny >= 0 && nx <= len && ny <= len && maps[nx][ny] === 1){
                    maps[nx][ny] = 0;
                    DFS(count + 1, nx, ny);
                    maps[nx][ny] = 1;
                }
                
            }
        }
    }
    
    maps[0][0] = 0;
    DFS(1, 0, 0);
    
    return answer
}
